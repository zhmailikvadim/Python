Symptom

Use Groovy to extract all ComposedType permissions (read, create, remove, change) for a user or usergroup.

Environment

commerce cloud all version

Resolution

Go to hac->console->script, use below groovy script:

import de.hybris.platform.servicelayer.security.permissions.PermissionCheckValue

def checkTypePermission(type, principal, permission) {

value = permissionCheckingService.checkTypePermission(type, principal, permission).getCheckValue()

if (value != PermissionCheckValue.NOT_DEFINED)

println principal.getUid() + "/" + type + "/" + permission + ": " + permissionCheckingService.checkTypePermission(type, principal, permission).getCheckValue()

}

def checkAttributePermission(attr, principal, permission) {

value = permissionCheckingService.checkAttributeDescriptorPermission(attr, principal, permission).getCheckValue()

if (value != PermissionCheckValue.NOT_DEFINED)

println principal.getUid() + "/" + attr.owner.code + "/" + attr.qualifier + "/" + permission + ": " + permissionCheckingService.checkAttributeDescriptorPermission(attr, principal, permission).getCheckValue()

}

userService = spring.getBean "userService"

permissionCheckingService = spring.getBean "permissionCheckingService"

flexibleSearchService = spring.getBean("flexibleSearchService")

principal = userService.getUserGroupForUID("customgroup") // extract permission for usergroup id

//principal = userService.getUserForUID("testc")  // extract permission for user id 


query = "SELECT {pk} FROM {ComposedType}"

result = flexibleSearchService.search(query)

result.result.each { type ->

checkTypePermission(type.code, principal, "read")

checkTypePermission(type.code, principal, "create")

checkTypePermission(type.code, principal, "change")

checkTypePermission(type.code, principal, "remove")

}

println "attribute permissions///////////////////////////////////////////////////////////////////////////////////////////////"

query = "SELECT {pk} FROM {AttributeDescriptor}"

result = flexibleSearchService.search(query)

result.result.each { attribute ->

checkAttributePermission(attribute, principal, "read")

checkAttributePermission(attribute, principal, "create")

checkAttributePermission(attribute, principal, "change")

checkAttributePermission(attribute, principal, "remove")

}
