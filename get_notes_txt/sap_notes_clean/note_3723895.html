Symptom

There is a requirement to export data from a single Commerce Cloud table.

Environment

Commerce Cloud

Resolution

Commerce Cloud provides several methods to export data. You should use the methods that best suits your use case.

  1. Backoffice Export Cronjob
    • You can configure an ImpEx export job to export the items you want to a Media zip file
  2. Export API
    • You can use the API to programmatically export items to a file
  3. Excel Export wizard
    • You can export data from the Backoffice to an Excel file via a wizard
  4. Export Through Administration Console
    • You can run an ImpEx Export script via the Administration Console
  5. Export Through Administration Console Scripting
    • If you have requirements which do not fit any of the above scenarios you can create a script to export the data you need
    • This can be used to export DB data which is not stored as items, such as Audit Logs stored in the DB or any other DB tables not accessible via ImpEx.

      As an example, the following script can be used to export the products1sn audit table data to a media file. The script is only provided as an example and can be customized to fit your specific needs.

      import java.sql.Connection
      import java.sql.PreparedStatement
      import java.sql.ResultSet
      import java.sql.SQLException
      import java.io.ByteArrayOutputStream
      import java.io.OutputStreamWriter
      import java.io.ByteArrayInputStream
      import java.text.SimpleDateFormat

      // =============================================================================
      // SAP Commerce Groovy Script: Export DB Table to Media File
      // =============================================================================

      // --- CONFIGURATION PARAMETERS ---
      def TABLE_NAME = "products1sn"                    // Target DB table to export
      def dateFormat = new SimpleDateFormat('yyyyMMdd_HHmmss')
      def MEDIA_CODE = "table_export_" + dateFormat.format(new Date())
      def FILE_NAME = "users_export.csv"
      def MIME_TYPE = "text/csv"
      def CATALOG_ID = "Default"
      def CATALOG_VERSION = "Staged"
      def CSV_DELIMITER = ","
      def CHARSET = "UTF-8"

      // --- GET SPRING BEANS ---
      def modelService = spring.getBean("modelService")
      def mediaService = spring.getBean("mediaService")
      def catalogService = spring.getBean("catalogService")
      def dataSource = spring.getBean("dataSource")

      Connection connection = null
      PreparedStatement statement = null
      ResultSet resultSet = null

      try {
          println "=== Starting DB Table Export ==="
          println "Table: ${TABLE_NAME}"
          println "Media Code: ${MEDIA_CODE}"
          
          // --- STEP 1: Query Database Table ---
          connection = dataSource.getConnection()
          def query = "SELECT * FROM ${TABLE_NAME}"
          
          statement = connection.prepareStatement(query)
          resultSet = statement.executeQuery()
          
          // Get column metadata
          def resultSetMetaData = resultSet.getMetaData()
          def columnCount = resultSetMetaData.getColumnCount()
          def columnNames = (1..columnCount).collect { i -> resultSetMetaData.getColumnName(i) }
          
          println "Found ${columnNames.size()} columns: ${columnNames}"
          
          // --- STEP 2: Generate CSV Content ---
          def csvContent = new ByteArrayOutputStream()
          def writer = new OutputStreamWriter(csvContent, CHARSET)
          
          // Write CSV header
          writer.write(columnNames.join(CSV_DELIMITER))
          writer.write("\n")
          
          // Write data rows
          def rowCount = 0
          while (resultSet.next()) {
              def values = (1..columnCount).collect { columnIndex ->
                  def value = resultSet.getObject(columnIndex)
                  if (value == null) {
                      return ""
                  }
                  
                  // Escape CSV special characters
                  def stringValue = value.toString()
                  if (stringValue.contains(CSV_DELIMITER) || 
                      stringValue.contains('"') || 
                      stringValue.contains('\n') || 
                      stringValue.contains('\r')) {
                      stringValue = '"' + stringValue.replace('"', '""') + '"'
                  }
                  return stringValue
              }
              
              writer.write(values.join(CSV_DELIMITER))
              writer.write("\n")
              rowCount++
          }
          
          writer.flush()
          writer.close()
          
          println "Exported ${rowCount} rows"
          
          if (rowCount == 0) {
              println "No data found in table ${TABLE_NAME}"
              return
          }
          
          // --- STEP 3: Get Catalog Version ---
          def catalog = catalogService.getCatalogForId(CATALOG_ID)
          if (!catalog) {
              throw new RuntimeException("Catalog not found: ${CATALOG_ID}")
          }
          
          def catalogVersion = catalogService.getCatalogVersion(CATALOG_ID, CATALOG_VERSION)
          if (!catalogVersion) {
              throw new RuntimeException("Catalog version not found: ${CATALOG_ID}:${CATALOG_VERSION}")
          }
          
          // --- STEP 4: Create Media Model ---
          def mediaModel = modelService.create(de.hybris.platform.core.model.media.MediaModel.class)
          mediaModel.setCode(MEDIA_CODE)
          mediaModel.setCatalogVersion(catalogVersion)
          mediaModel.setRealFileName(FILE_NAME)
          mediaModel.setMime(MIME_TYPE)
          mediaModel.setDescription("Database export of table: ${TABLE_NAME}")
          
          modelService.save(mediaModel)
          println "Created media model with code: ${MEDIA_CODE}"
          
          // --- STEP 5: Upload CSV Data to Media ---
          def csvBytes = csvContent.toByteArray()
          
          // Use the correct signature: setDataForMedia(MediaModel, byte[])
          mediaService.setDataForMedia(mediaModel, csvBytes)
          
          println "=== Export Completed Successfully ==="
          println "Media PK: ${mediaModel.getPk()}"
          println "File size: ${csvBytes.length} bytes"
          println "Records exported: ${rowCount}"
          
          return mediaModel
          
      } catch (SQLException e) {
          println "SQL Error: ${e.message}"
          e.printStackTrace()
          throw e
      } catch (Exception e) {
          println "Error during export: ${e.message}"
          e.printStackTrace()
          throw e
      } finally {
          // Clean up resources
          try { resultSet?.close() } catch (ignored) {}
          try { statement?.close() } catch (ignored) {}
          try { connection?.close() } catch (ignored) {}
      }
