Symptom

How to schedule report when a custom event is manually triggered using Java SDK ?

Environment

SAP BusinessObjects Business Intelligence platform 4.x

Resolution

  • Use the jsp code below :

<%@ page import="com.crystaldecisions.sdk.exception.SDKException" %>
<%@ page import="com.crystaldecisions.sdk.framework.CrystalEnterprise" %>
<%@ page import="com.crystaldecisions.sdk.framework.IEnterpriseSession" %>
<%@ page import="com.crystaldecisions.sdk.occa.infostore.CeScheduleType" %>
<%@ page import="com.crystaldecisions.sdk.occa.infostore.IEvents" %>
<%@ page import="com.crystaldecisions.sdk.occa.infostore.IInfoObject" %>
<%@ page import="com.crystaldecisions.sdk.occa.infostore.IInfoObjects" %>
<%@ page import="com.crystaldecisions.sdk.occa.infostore.IInfoStore" %>
<%@ page import="com.crystaldecisions.sdk.occa.infostore.ISchedulingInfo" %>
<%@ page import="com.crystaldecisions.sdk.plugin.desktop.event.IEvent" %>
<%@ page import="com.crystaldecisions.sdk.properties.IProperties" %>

<html>
<head>
    <title>Schedule Report when a custom event is manually triggered</title>
</head>
<body>
<%
    IEnterpriseSession es = null;
    try {
        String user = "administrator";
        String password = "******"; 
        String cmsName = "*****";
        String cmsAuthType = "secEnterprise";

        // Log in to the CMS
        es = CrystalEnterprise.getSessionMgr().logon(user, password, cmsName, cmsAuthType);

        IInfoStore iStore = (IInfoStore) es.getService("InfoStore");

        // Query for the specific report by ID
        String reportId = "*****"; // replace with your report ID
        IInfoObjects oInfoObjects = iStore.query("SELECT * FROM CI_INFOOBJECTS WHERE SI_ID=" + reportId);

        if (oInfoObjects.size() > 0) {
            IInfoObject oInfoObject = (IInfoObject) oInfoObjects.get(0);

            try {
                // Display the report title
                out.println("<p>Report Name: " + oInfoObject.getTitle() + "</p>");

                // Get scheduling information
                ISchedulingInfo schedulingInfo = oInfoObject.getSchedulingInfo();
                IProperties properties = schedulingInfo.properties();

                // Query for the event by ID , Make sure to enter EVENT ID
                IInfoObjects events = iStore.query("SELECT * FROM CI_SYSTEMOBJECTS WHERE SI_ID=****");
                if (events.size() == 0) {
                    throw new RuntimeException("Event 'Schedule Event Test' not found!");
                }
                IEvent event = (IEvent) events.get(0);

                // Configure scheduling
                schedulingInfo.setRightNow(true);
                schedulingInfo.setType(CeScheduleType.ONCE);

                // Add event dependency
                IEvents scheduleEvents = schedulingInfo.getDependencies();
                int eventID = event.getID();
                scheduleEvents.add(eventID);

                // Schedule the report
                iStore.schedule(oInfoObjects);

                // Display the new schedule time
                out.println("<p>The report is scheduled at: "
                        + properties.getProperty("SI_STARTTIME").getValue() + "</p>");
            } catch (Exception e) {
                out.println("<p style='color:red;'>An error occurred: " + e.getMessage() + "</p>");
            }
        } else {
            out.println("<p style='color:red;'>Report with ID " + reportId + " not found.</p>");
        }

    } catch (SDKException e) {
        out.println("<p style='color:red;'>SDK Error: " + e.getMessage() + "</p>");
        e.printStackTrace();
    } finally {
        if (es != null) {
            es.logoff();
        }
    }
%>
</body>
</html>

  • To run the code, refer to the steps in KBA 2932452 - Steps to Create / Deploy Web Application in Tomcat for BI4.x Java SDK ?
